Find the Kth Max and Min Element in an Array

Solution Explanation

Chalo samjhte hain kaise find karte hain Kth maximum aur minimum element ek array mein step by step:

Method 1: Using Sorting

Step 1: Pehle ek array define karte hain:

int[] array = {3, 5, 1, 8, 2};
int K = 2;

Step 2: Array ko sort karte hain:

Arrays.sort(array);

Step 3: Kth minimum aur Kth maximum element ko find karte hain:

int KthMin = array[K - 1];
int KthMax = array[array.length - K];

Step 4: Result ko print karte hain:

System.out.println("Kth Max: " + KthMax + ", Kth Min: " + KthMin);

Method 2: Using Priority Queue

Step 1: Pehle ek array define karte hain:

int[] array = {3, 5, 1, 8, 2};
int K = 2;

Step 2: MinHeap aur MaxHeap ko initialize karte hain:

PriorityQueue minHeap = new PriorityQueue<>();
PriorityQueue maxHeap = new PriorityQueue<>(Collections.reverseOrder());

Step 3: Array ke elements ko heaps mein add karte hain:

for (int num : array) {
    minHeap.add(num);
    maxHeap.add(num);
}

Step 4: Kth minimum aur Kth maximum element ko find karte hain:

int KthMin = 0, KthMax = 0;
for (int i = 0; i < K; i++) {
    KthMin = minHeap.poll();
    KthMax = maxHeap.poll();
}

Step 5: Result ko print karte hain:

System.out.println("Kth Max: " + KthMax + ", Kth Min: " + KthMin);

Method 3: Using Quickselect

Step 1: Pehle ek array define karte hain:

int[] array = {3, 5, 1, 8, 2};
int K = 2;

Step 2: Quickselect function ko call karte hain:

int KthMin = quickselect(array, 0, array.length - 1, K - 1);
int KthMax = quickselect(array, 0, array.length - 1, array.length - K);

Step 3: Quickselect function ko define karte hain:

public static int quickselect(int[] array, int left, int right, int K) {
    if (left == right) {
        return array[left];
    }
    int pivotIndex = partition(array, left, right);
    if (K == pivotIndex) {
        return array[K];
    } else if (K < pivotIndex) {
        return quickselect(array, left, pivotIndex - 1, K);
    } else {
        return quickselect(array, pivotIndex + 1, right, K);
    }
}

Step 4: Partition function ko define karte hain:

public static int partition(int[] array, int left, int right) {
    int pivot = array[right];
    int i = left;
    for (int j = left; j < right; j++) {
        if (array[j] <= pivot) {
            int temp = array[i];
            array[i] = array[j];
            array[j] = temp;
            i++;
        }
    }
    int temp = array[i];
    array[i] = array[right];
    array[right] = temp;
    return i;
}

Step 5: Result ko print karte hain:

System.out.println("Kth Max: " + KthMax + ", Kth Min: " + KthMin);

public static int[] findKthMaxMin(int[] array, int K) {
    // MinHeap aur MaxHeap ko initialize karte hain
    PriorityQueue minHeap = new PriorityQueue<>();
    PriorityQueue maxHeap = new PriorityQueue<>(Collections.reverseOrder());
    // Array ke elements ko heaps mein add karte hain
    for (int num : array) {
        minHeap.add(num);
        maxHeap.add(num);
    }
    // Kth minimum aur Kth maximum element ko find karte hain
    int KthMin = 0, KthMax = 0;
    for (int i = 0; i < K; i++) {
        KthMin = minHeap.poll();
        KthMax = maxHeap.poll();
    }
    return new int[]{KthMax, KthMin};
}

import heapq

def find_kth_max_min(arr, K):
    # Kth minimum aur Kth maximum element ko find karte hain using heapq
    Kth_min = heapq.nsmallest(K, arr)[-1]
    Kth_max = heapq.nlargest(K, arr)[-1]
    return Kth_max
